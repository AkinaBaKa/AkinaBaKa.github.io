### 2024.06.11 安全的密码/凯撒密码
```
#include <stdio.h>
#include <string.h>
int cirno (char*);
int main(void) 
{
	int num;
	scanf("%d",&num);
	getchar();
	while(num-->0)
	{
		char baka[100]={0};
		gets(baka);
		if(strlen(baka)<6)
			printf("Not Safe\n");
		else
			cirno(baka);
	}
	return 0;
}
int cirno (char* busu)
{
	int nu=0,en=0,es=0,caps=0;
	for(int n=0;n<strlen(busu);n++)
	{
		if((busu[n]>='!'&&busu[n]<='/')||(busu[n]>=':'&&busu[n]<='@')||(busu[n]>='['&&busu[n]<='`')||(busu[n]==' '))
			es=1;
		else if(busu[n]>='0'&&busu[n]<='9')
			nu=1;
		else if(busu[n]>='a'&&busu[n]<='z')
			en=1;
		else if((busu[n]>='A'&&busu[n]<='Z'))
			caps=1;
	}
	//printf("%d %d %d\n",nu,en,es);
	if(nu+en+es+caps==1)
		printf("Not Safe\n");
	else if(nu+en+es+caps==2)
		printf("Medium Safe\n");
	else if(nu+en+es+caps>=3)
		printf("Safe\n");
	return 0;
}
```
