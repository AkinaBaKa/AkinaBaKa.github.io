**2024.06.25 北理工乐学题目 图形题 打印字母"G"**
```
#include <stdio.h>
int main()
{
	int num;
	scanf("%d",&num);
	for(int n=0;n<2*num-1;n++)
		printf("G");
	printf("\n");
	for(int y=0;y<num-2;y++)
	{
		printf("G");
		for(int n=0;n<2*num-2;n++)
			printf("+");
		printf("\n");
	}
	printf("G");
	for(int n=0;n<2*num-2-num;n++)
		printf("+");
	for(int n=0;n<num;n++)
		printf("G");
	printf("\n");
	for(int y=0;y<num-2;y++)
	{
		printf("G");
		for(int n=0;n<2*num-3;n++)
			printf("+");
		printf("G\n");
	}
	for(int n=0;n<2*num-1;n++)
		printf("G");
	printf("\n");
	return 0;
}
```
